export function useTranslation() {
  const { language } = useSettings();

  const t = (key, vars = {}) => {
    const keys = key.split('.');
    let result = translations[language];
    for (const k of keys) {
      result = result?.[k];
    }
    if (result === undefined) {
      result = translations['EN'];
      for (const k of keys) {
        result = result?.[k];
      }
    }
    if (typeof result !== 'string') return key;
    return Object.entries(vars).reduce(
      (str, [k, v]) => str.replace(`{${k}}`, v),
      result
    );
  };

  return { t };
}

## Interpolation

Variables are interpolated with `{varName}` syntax:
- `t('dashboard.welcome', { username })` → "Welcome back, John! 👋"
- `t('posts.resultCount', { filtered: 3, total: 10 })` → "3 of 10 posts"
- `t('calendar.postsThisMonth', { count: 5 })` → "5 posts this month"
